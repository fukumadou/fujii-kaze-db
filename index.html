// 検索機能の続きと強化 searchInput.addEventListener('input', (e) => { const searchTerm = e.target.value.toLowerCase(); // ✕ボタンの表示切り替え searchClearBtn.style.display = searchTerm ? 'block' : 'none'; let hasAnyMatch = false; accordionItems.forEach(item => { const trackItems = item.querySelectorAll('.track-item'); let hasTrackMatch = false; trackItems.forEach(track => { const trackName = track.getAttribute('data-name'); if (trackName.includes(searchTerm)) { track.style.display = 'flex'; hasTrackMatch = true; hasAnyMatch = true; } else { track.style.display = 'none'; } }); // 検索にヒットした楽曲があれば、アコーディオンを開く if (hasTrackMatch && searchTerm !== "") { item.classList.add('active'); item.querySelector('.accordion-content').style.maxHeight = item.querySelector('.accordion-content').scrollHeight + "px"; } else if (searchTerm === "") { // 文字が空なら閉じる item.classList.remove('active'); item.querySelector('.accordion-content').style.maxHeight = null; } }); noResults.style.display = hasAnyMatch ? 'none' : 'block'; }); // ✕ボタンのイベント searchClearBtn.addEventListener('click', resetSearch);
